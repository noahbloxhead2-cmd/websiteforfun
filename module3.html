Shortcut Manager Module

(Made for the creator — not really for public use)
const win0 = makeWindow(200,500,'Shortcut Manager',0,0); const optionContainer = document.createElement('div'); win0.appendChild(optionContainer); const createShortcutButton = document.createElement('button'); createShortcutButton.textContent = 'Create'; optionContainer.appendChild(createShortcutButton); const viewShortcutButton = document.createElement('button'); viewShortcutButton.textContent = 'View'; optionContainer.appendChild(viewShortcutButton); const removeShortcutButton = document.createElement('button'); removeShortcutButton.textContent = 'Delete'; optionContainer.appendChild(removeShortcutButton); createShortcutButton.onclick = function() { const winC = makeWindow(200,500,'Create Shortcut',0,0); const shortcutInp = document.createElement('textarea'); shortcutInp.placeholder = "Shortcut here..."; winC.appendChild(shortcutInp); const linkInp = document.createElement('textarea'); linkInp.placeholder = "Url here..."; winC.appendChild(linkInp); const createButton = document.createElement('button'); createButton.textContent = "Make shortcut"; winC.appendChild(createButton); createButton.onclick = function() { shortcuts[shortcutInp.value] = linkInp.value; shortcutInp.value = "Shortcut created."; linkInp.value = "Text goes here"; } } viewShortcutButton.onclick = function() { const winV = makeWindow(200,500,'View Shortcuts',0,0); const scrollBar = document.createElement('div'); scrollBar.style.overflow = 'scroll'; winV.appendChild(scrollBar); for (const key in shortcuts) { const shortcutTxt = document.createElement('h1'); shortcutTxt.textContent = `${key}: ${shortcuts[key]}`; shortcutTxt.style.color = 'black'; scrollBar.appendChild(shortcutTxt); } } removeShortcutButton.onclick = function() { const winV = makeWindow(80,500,'Remove Shortcuts',0,0); const scrollBar = document.createElement('div'); scrollBar.style.overflow = 'scroll'; winV.appendChild(scrollBar); for (const key in shortcuts) { const shortcutTxt = document.createElement('h1'); shortcutTxt.textContent = key; shortcutTxt.style.color = 'black'; scrollBar.appendChild(shortcutTxt); const removebutton = document.createElement('button'); removebutton.textContent = 'Delete'; scrollBar.appendChild(removebutton); removebutton.onclick = function() { delete shortcuts[key]; shortcutTxt.textContent = 'Deleted'; shortcutTxt.style.color = 'red'; } } }
