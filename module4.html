const win0 = makeWindow(200,400,"WinNet",0,0); const searchBar = document.createElement('textarea'); searchBar.placeholder = 'Search the somewhat wide web'; win0.appendChild(searchBar); const searchButton = document.createElement('button'); searchButton.textContent = "Search"; win0.appendChild(searchButton); const sizeInputs = document.createElement('textarea'); sizeInputs.value = "500 1380"; win0.appendChild(sizeInputs); searchButton.onclick = function() { try { const url = searchBar.value; const sizes = sizeInputs.value.split(" "); const resultWin = makeWindow(`${sizes[0]}`,`${sizes[1]}`,`${url} -- WinNet`,0,0); const web = document.createElement('iframe'); web.style.height = `${sizes[0]}px`; web.style.width = `${sizes[1]}px`; web.src = `https://${url}`; resultWin.appendChild(web); } catch (e) { searchBar.value = `Error: ${e.message}` } }
